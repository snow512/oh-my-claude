# Gist-Based Settings Sync Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Enable cross-machine sync of oh-my-claude settings/skills via GitHub Gist as a personal cloud store.

**Architecture:** New `src/sync.ts` module handles all Gist API communication and sync logic. Three new CLI commands (`login`, `push`, `pull`) route through `src/cli.ts`. Auth credentials stored in `~/.claude/.omc-auth`. Sync compares local state against Gist manifest to detect changes, uploads modified skills as flat files (prefix `omc-skill--`), and downloads/applies them on pull.

**Tech Stack:** Node.js built-in `https` module (zero dependencies), GitHub Gist REST API v3, JSON manifest for state tracking.

```typescript
import * as fs from 'fs';
import * as path from 'path';
import * as https from 'https';
import { CLAUDE_DIR, PACKAGE_ROOT, readJson, writeJson, isDirChanged } from './installer';
import { renderBanner, renderStep, progressLine, ask, C, style } from './ui';
import type { Opts } from './installer';

// --- Types ---

interface AuthData {
  token: string;
  gistId?: string;
}

interface SyncManifest {
  version: string;
  timestamp: string;
  skills: {
    installed: string[];
    removed: string[];
    modified: string[];
    custom: string[];
  };
  lang: string;
}

interface GistFile {
  content: string;
}

interface GistPayload {
  description: string;
  public: boolean;
  files: Record<string, GistFile | null>;
}

interface GistResponse {
  id: string;
  html_url: string;
  files: Record<string, { content: string }>;
}

// --- Constants ---

const AUTH_PATH = path.join(CLAUDE_DIR, '.omc-auth');
const GIST_PREFIX = 'omc-skill--';
const MANIFEST_FILE = 'omc-manifest.json';
const SETTINGS_FILE = 'omc-settings.json';
const CLAUDE_MD_FILE = 'omc-claude-md.md';

// --- Auth ---

function loadAuth(): AuthData | null {
  try {
    return JSON.parse(fs.readFileSync(AUTH_PATH, 'utf-8'));
  } catch {
    return null;
  }
}

function saveAuth(data: AuthData): void {
  fs.mkdirSync(path.dirname(AUTH_PATH), { recursive: true });
  fs.writeFileSync(AUTH_PATH, JSON.stringify(data, null, 2) + '\n');
  fs.chmodSync(AUTH_PATH, 0o600);
}

// --- GitHub API ---

function githubApi(method: string, endpoint: string, token: string, body?: unknown): Promise<GistResponse> {
  return new Promise((resolve, reject) => {
    const data = body ? JSON.stringify(body) : undefined;
    const req = https.request({
      hostname: 'api.github.com',
      path: endpoint,
      method,
      headers: {
        'Authorization': `Bearer ${token}`,
        'Accept': 'application/vnd.github+json',
        'User-Agent': 'oh-my-claude',
        'X-GitHub-Api-Version': '2022-11-28',
        ...(data ? { 'Content-Type': 'application/json', 'Content-Length': Buffer.byteLength(data) } : {}),
      },
    }, (res) => {
      let body = '';
      res.on('data', (chunk: Buffer) => body += chunk);
      res.on('end', () => {
        if (res.statusCode && res.statusCode >= 200 && res.statusCode < 300) {
          try { resolve(JSON.parse(body)); }
          catch { reject(new Error(`Invalid JSON response: ${body.slice(0, 100)}`)); }
        } else {
          reject(new Error(`GitHub API ${res.statusCode}: ${body.slice(0, 200)}`));
        }
      });
    });
    req.on('error', reject);
    if (data) req.write(data);
    req.end();
  });
}

// --- Manifest ---

function detectLang(): string {
  const skillsDest = path.join(CLAUDE_DIR, 'skills');
  try {
    const first = fs.readdirSync(skillsDest, { withFileTypes: true }).find(e => e.isDirectory());
    if (first) {
      const content = fs.readFileSync(path.join(skillsDest, first.name, 'SKILL.md'), 'utf-8');
      if (/[\uAC00-\uD7AF]/.test(content.split('---')[2] || '')) return 'ko';
    }
  } catch {}
  return 'en';
}

function buildManifest(): { manifest: SyncManifest; modifiedFiles: Record<string, string> } {
  const skillsSrc = path.join(PACKAGE_ROOT, 'user-skills');
  const skillsDest = path.join(CLAUDE_DIR, 'skills');

  const repoSkills = new Set<string>();
  try {
    for (const e of fs.readdirSync(skillsSrc, { withFileTypes: true })) {
      if (e.isDirectory()) repoSkills.add(e.name);
    }
  } catch {}

  const localSkills = new Set<string>();
  try {
    for (const e of fs.readdirSync(skillsDest, { withFileTypes: true })) {
      if (e.isDirectory()) localSkills.add(e.name);
    }
  } catch {}

  const installed: string[] = [];
  const removed: string[] = [];
  const modified: string[] = [];
  const custom: string[] = [];
  const modifiedFiles: Record<string, string> = {};

  for (const name of repoSkills) {
    if (!localSkills.has(name)) {
      removed.push(name);
    } else if (isDirChanged(path.join(skillsSrc, name), path.join(skillsDest, name))) {
      modified.push(name);
      try {
        const content = fs.readFileSync(path.join(skillsDest, name, 'SKILL.md'), 'utf-8');
        modifiedFiles[`${GIST_PREFIX}${name}.md`] = content;
      } catch {}
    } else {
      installed.push(name);
    }
  }

  for (const name of localSkills) {
    if (!repoSkills.has(name)) {
      custom.push(name);
      try {
        const content = fs.readFileSync(path.join(skillsDest, name, 'SKILL.md'), 'utf-8');
        modifiedFiles[`${GIST_PREFIX}${name}.md`] = content;
      } catch {}
    }
  }

  const manifest: SyncManifest = {
    version: require(path.join(PACKAGE_ROOT, 'package.json')).version,
    timestamp: new Date().toISOString(),
    skills: { installed, removed, modified, custom },
    lang: detectLang(),
  };

  return { manifest, modifiedFiles };
}

// --- Commands ---

export async function runLogin(opts: Opts = {}): Promise<void> {
  renderBanner();
  console.log(`  ${style('GitHub Token Setup', C.bold)}\n`);
  console.log(`  ${style('Generate a token at:', C.gray)} https://github.com/settings/tokens`);
  console.log(`  ${style('Required scope:', C.gray)} gist\n`);

  const existing = loadAuth();
  if (existing?.token) {
    const masked = existing.token.slice(0, 7) + '...' + existing.token.slice(-4);
    console.log(`  ${style('Current token:', C.gray)} ${masked}`);
    if (!opts.force && !await ask('Replace existing token?', false)) {
      console.log(`  ${style('⏭', C.gray)}  Kept existing token\n`);
      return;
    }
  }

  const rl = readline.createInterface({ input: process.stdin, output: process.stdout });
  const token = await new Promise<string>((resolve) => {
    rl.question(`  ${style('Token:', C.cyan)} `, (answer) => {
      rl.close();
      resolve(answer.trim());
    });
  });

  if (!token) {
    console.log(`\n  ${style('✗', C.red)} No token provided\n`);
    return;
  }

  // Validate token
  console.log('');
  try {
    await progressLine('Verifying token', async () => {
      await githubApi('GET', '/user', token);
    });
  } catch {
    console.log(`  ${style('✗', C.red)} Invalid token or network error\n`);
    return;
  }

  saveAuth({ token, gistId: existing?.gistId });
  console.log(`  ${style('✓', C.green)} ${style('Token saved', C.bold)}\n`);
}

export async function runPush(args: string[] | undefined, opts: Opts = {}): Promise<void> {
  renderBanner();
  console.log(`  ${style('Pushing settings to cloud...', C.bold)}\n`);

  const auth = loadAuth();
  if (!auth?.token) {
    console.log(`  ${style('✗', C.red)} Not logged in. Run ${style('omc login', C.cyan)} first.\n`);
    return;
  }

  // Build manifest and detect changes
  const { manifest, modifiedFiles } = buildManifest();

  // If specific skills provided, filter to only those
  const specificSkills = args?.filter(a => !a.startsWith('-'));
  if (specificSkills && specificSkills.length > 0) {
    const allChanged = [...manifest.skills.modified, ...manifest.skills.custom];
    const filtered = specificSkills.filter(s => allChanged.includes(s));
    if (filtered.length === 0) {
      console.log(`  ${style('–', C.gray)} Specified skills have no changes\n`);
      return;
    }
    // Filter modifiedFiles to only specified skills
    for (const key of Object.keys(modifiedFiles)) {
      const skillName = key.replace(GIST_PREFIX, '').replace('.md', '');
      if (!filtered.includes(skillName)) delete modifiedFiles[key];
    }
    manifest.skills.modified = manifest.skills.modified.filter(s => filtered.includes(s));
    manifest.skills.custom = manifest.skills.custom.filter(s => filtered.includes(s));
  }

  // Show summary
  const { skills } = manifest;
  const totalChanges = skills.modified.length + skills.custom.length + skills.removed.length;

  console.log(`  ${style('Local state:', C.bold)}`);
  console.log(`    ${style('–', C.gray)} ${skills.installed.length} skills (unchanged)`);
  if (skills.modified.length > 0) console.log(`    ${style('!', C.yellow)} ${skills.modified.length} modified: ${skills.modified.join(', ')}`);
  if (skills.custom.length > 0) console.log(`    ${style('+', C.green)} ${skills.custom.length} custom: ${skills.custom.join(', ')}`);
  if (skills.removed.length > 0) console.log(`    ${style('✗', C.red)} ${skills.removed.length} removed: ${skills.removed.join(', ')}`);
  console.log('');

  if (totalChanges === 0) {
    console.log(`  ${style('–', C.gray)} No changes to push\n`);
    return;
  }

  if (!opts.yes && !await ask(`Push ${totalChanges} change(s) to cloud?`, true)) {
    console.log(`  ${style('⏭', C.gray)}  Cancelled\n`);
    return;
  }

  // Build Gist files
  const gistFiles: Record<string, GistFile | null> = {
    [MANIFEST_FILE]: { content: JSON.stringify(manifest, null, 2) },
  };

  // Add settings
  const settingsPath = path.join(CLAUDE_DIR, 'settings.json');
  if (fs.existsSync(settingsPath)) {
    const settings = readJson(settingsPath);
    if (settings) {
      // Only sync omc-relevant keys
      const syncKeys = ['permissions', 'enabledPlugins', 'extraKnownMarketplaces'];
      const filtered: Record<string, unknown> = {};
      for (const key of syncKeys) {
        if (key in (settings as Record<string, unknown>)) {
          filtered[key] = (settings as Record<string, unknown>)[key];
        }
      }
      gistFiles[SETTINGS_FILE] = { content: JSON.stringify(filtered, null, 2) };
    }
  }

  // Add CLAUDE.md omc block
  const claudeMdPath = path.join(CLAUDE_DIR, 'CLAUDE.md');
  if (fs.existsSync(claudeMdPath)) {
    const content = fs.readFileSync(claudeMdPath, 'utf-8');
    const omcStart = '<!-- <omc>';
    const omcEnd = '<!-- </omc> -->';
    const startIdx = content.indexOf(omcStart);
    const endIdx = content.indexOf(omcEnd);
    if (startIdx !== -1 && endIdx !== -1) {
      gistFiles[CLAUDE_MD_FILE] = { content: content.slice(startIdx, endIdx + omcEnd.length) };
    }
  }

  // Add modified/custom skill files
  for (const [filename, content] of Object.entries(modifiedFiles)) {
    gistFiles[filename] = { content };
  }

  // Create or update Gist
  try {
    let gist: GistResponse;
    if (auth.gistId) {
      await progressLine('Updating cloud sync', async () => {
        gist = await githubApi('PATCH', `/gists/${auth.gistId}`, auth.token, {
          description: 'oh-my-claude sync',
          files: gistFiles,
        } as GistPayload);
      });
      console.log(`  ${style('✓', C.green)} ${style('Sync updated', C.bold)}`);
    } else {
      await progressLine('Creating cloud sync', async () => {
        gist = await githubApi('POST', '/gists', auth.token, {
          description: 'oh-my-claude sync',
          public: false,
          files: gistFiles,
        } as GistPayload);
        auth.gistId = gist!.id;
        saveAuth(auth);
      });
      console.log(`  ${style('✓', C.green)} ${style('Sync created', C.bold)}`);
    }
    console.log(`  ${style('Gist:', C.gray)} ${auth.gistId}\n`);
  } catch (err) {
    console.log(`  ${style('✗', C.red)} Push failed: ${(err as Error).message}\n`);
  }
}

export async function runPull(opts: Opts = {}): Promise<void> {
  renderBanner();
  console.log(`  ${style('Pulling settings from cloud...', C.bold)}\n`);

  const auth = loadAuth();
  if (!auth?.token || !auth.gistId) {
    if (!auth?.token) {
      console.log(`  ${style('✗', C.red)} Not logged in. Run ${style('omc login', C.cyan)} first.\n`);
    } else {
      console.log(`  ${style('✗', C.red)} No sync found. Run ${style('omc push', C.cyan)} first.\n`);
    }
    return;
  }

  // Fetch Gist
  let gist: GistResponse;
  try {
    await progressLine('Fetching cloud sync', async () => {
      gist = await githubApi('GET', `/gists/${auth.gistId}`, auth.token);
    });
  } catch (err) {
    console.log(`  ${style('✗', C.red)} Pull failed: ${(err as Error).message}\n`);
    return;
  }

  const files = gist!.files;

  // Parse manifest
  if (!files[MANIFEST_FILE]) {
    console.log(`  ${style('✗', C.red)} Invalid sync data (no manifest)\n`);
    return;
  }

  const manifest: SyncManifest = JSON.parse(files[MANIFEST_FILE].content);
  console.log(`  ${style('Synced at:', C.gray)} ${manifest.timestamp}`);
  console.log(`  ${style('Language:', C.gray)} ${manifest.lang}\n`);

  // Apply settings
  if (files[SETTINGS_FILE]) {
    const remoteSettings = JSON.parse(files[SETTINGS_FILE].content);
    const settingsPath = path.join(CLAUDE_DIR, 'settings.json');
    const local = readJson(settingsPath) || {};

    const merged = { ...(local as Record<string, unknown>) };
    for (const [key, value] of Object.entries(remoteSettings)) {
      merged[key] = value;
    }

    if (opts.yes || await ask('Apply synced permissions & plugins?', true)) {
      backup(settingsPath);
      writeJson(settingsPath, merged);
      console.log(`  ${style('✓', C.green)} Settings applied`);
    } else {
      console.log(`  ${style('⏭', C.gray)}  Settings skipped`);
    }
  }

  // Apply CLAUDE.md
  if (files[CLAUDE_MD_FILE]) {
    const claudeMdPath = path.join(CLAUDE_DIR, 'CLAUDE.md');
    const existing = fs.existsSync(claudeMdPath) ? fs.readFileSync(claudeMdPath, 'utf-8') : '';
    const omcStart = '<!-- <omc>';
    const omcEnd = '<!-- </omc> -->';
    const remoteBlock = files[CLAUDE_MD_FILE].content;

    if (existing.includes(omcStart)) {
      const startIdx = existing.indexOf(omcStart);
      const endIdx = existing.indexOf(omcEnd);
      if (startIdx !== -1 && endIdx !== -1) {
        const currentBlock = existing.slice(startIdx, endIdx + omcEnd.length);
        if (currentBlock !== remoteBlock) {
          if (opts.yes || await ask('Update CLAUDE.md omc section?', true)) {
            const updated = existing.replace(currentBlock, remoteBlock);
            fs.writeFileSync(claudeMdPath, updated);
            console.log(`  ${style('✓', C.green)} CLAUDE.md updated`);
          }
        } else {
          console.log(`  ${style('–', C.gray)} CLAUDE.md (up to date)`);
        }
      }
    } else if (opts.yes || await ask('Add omc section to CLAUDE.md?', true)) {
      const newContent = existing ? existing.trimEnd() + '\n\n' + remoteBlock + '\n' : remoteBlock + '\n';
      fs.mkdirSync(path.dirname(claudeMdPath), { recursive: true });
      fs.writeFileSync(claudeMdPath, newContent);
      console.log(`  ${style('✓', C.green)} CLAUDE.md updated`);
    }
  }

  // Apply skills
  const skillsDest = path.join(CLAUDE_DIR, 'skills');
  fs.mkdirSync(skillsDest, { recursive: true });

  // Removed skills
  if (manifest.skills.removed.length > 0) {
    console.log(`\n  ${style('Removed skills:', C.bold)}`);
    for (const name of manifest.skills.removed) {
      const destPath = path.join(skillsDest, name);
      if (fs.existsSync(destPath)) {
        if (opts.yes || await ask(`  Remove ${name}?`, true)) {
          fs.rmSync(destPath, { recursive: true });
          console.log(`    ${style('✗', C.red)} ${name} removed`);
        }
      }
    }
  }

  // Modified/custom skills
  const skillFiles = Object.keys(files).filter(f => f.startsWith(GIST_PREFIX));
  if (skillFiles.length > 0) {
    console.log(`\n  ${style('Synced skills:', C.bold)}`);
    for (const filename of skillFiles) {
      const skillName = filename.replace(GIST_PREFIX, '').replace('.md', '');
      const destDir = path.join(skillsDest, skillName);
      fs.mkdirSync(destDir, { recursive: true });
      fs.writeFileSync(path.join(destDir, 'SKILL.md'), files[filename].content);
      console.log(`    ${style('✓', C.green)} ${skillName}`);
    }
  }
```

```typescript
// --- Test: Pull merge strategy ---

describe('pull merge strategy', () => {
  it('should merge remote settings into local without losing non-omc keys', () => {
    const local = {
      statusLine: 'custom-script.sh',
      permissions: { allow: ['old-rule'] },
      customKey: true,
    };

    const remote = {
      permissions: { allow: ['new-rule-1', 'new-rule-2'], deny: ['deny-1'] },
      enabledPlugins: { 'plugin-a': true },
    };

    const merged = { ...local } as Record<string, unknown>;
    for (const [key, value] of Object.entries(remote)) {
      merged[key] = value;
    }

    // Remote keys overwrite
    assert.deepEqual(merged.permissions, remote.permissions);
    assert.deepEqual(merged.enabledPlugins, remote.enabledPlugins);
    // Local-only keys preserved
    assert.equal(merged.statusLine, 'custom-script.sh');
    assert.equal(merged.customKey, true);
  });
});
```
